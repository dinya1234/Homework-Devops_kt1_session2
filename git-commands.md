"# Git Commands Cheatsheet ## Basic Commands \`\`\`bash # Initialize a new repository git init # Clone a repository git clone <repository-url> # Check status git status # Add files to staging git add <filename> git add .          # Add all files git add -A         # Add all files including deletions \`\`\` ## Commit Commands \`\`\`bash # Create a commit git commit -m \"Your commit message\" # Amend last commit git commit --amend -m \"New message\" # Add files to last commit git add . git commit --amend --no-edit \`\`\` ## Branch Commands \`\`\`bash # Create and switch to new branch git checkout -b <branch-name> # Switch to existing branch git checkout <branch-name> # List all branches git branch -a # List local branches git branch # Delete branch locally git branch -d <branch-name> # Delete branch remotely git push origin --delete <branch-name> \`\`\` ## View History \`\`\`bash # View commit history git log # View compact history git log --oneline --graph --all # View history of specific file git log --oneline <filename> # View tags history git log --tags --simplify-by-decoration --pretty=\"format:%ai %d\" \`\`\` ## Moving Changes Between Branches \`\`\`bash # Cherry-pick specific commit git cherry-pick <commit-hash> # Apply changes from another branch git checkout <source-branch> -- <file1> <file2> # Stash changes and apply to another branch git stash git checkout <other-branch> git stash pop \`\`\` ## Remote Operations \`\`\`bash # Push to remote git push origin <branch-name> # Pull from remote git pull origin <branch-name> # Fetch updates git fetch --all # View remotes git remote -v # Add remote git remote add <name> <url> \`\`\` ## Tags \`\`\`bash # Create tag git tag v1.0.0 # Create annotated tag git tag -a v1.0.0 -m \"Version 1.0.0 release\" # Push tags to remote git push origin --tags # View all tags git tag -l # View tag history git log --tags --simplify-by-decoration --pretty=\"format:%ai %d\" \`\`\` ## Useful Aliases \`\`\`bash # Add to ~/.gitconfig [alias] co = checkout br = branch ci = commit st = status lg = log --oneline --graph --all last = log -1 HEAD unstage = reset HEAD -- \`\`\` ## Help \`\`\`bash # Get help on any command git help <command> # Example: git help commit \`\`\`"
